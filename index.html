document.getElementById('recipeForm').addEventListener('submit', async (e) => { e.preventDefault(); const title = document.getElementById('recipeTitle').value; const content = document.getElementById('recipeContent').value; const imageFile = document.getElementById('recipeImage').files[0]; let uploadedImageUrl = null; // 1. Gestion de l'image optionnelle if (imageFile) { const fileName = `${Date.now()}_${imageFile.name}`; try { const { error: upError } = await supabaseClient.storage.from('recipe-images').upload(fileName, imageFile); if (!upError) { const { data } = supabaseClient.storage.from('recipe-images').getPublicUrl(fileName); uploadedImageUrl = data.publicUrl; } } catch(err) { console.log("Erreur stockage image"); } } // 2. Vérification de la session utilisateur if (!currentUserEmail) { alert("Erreur : Vous devez être connecté pour ajouter une recette."); return; } // 3. Insertion stricte avec l'auteur connecté const { error } = await supabaseClient.from('recipes').insert([{ title, content, image_url: uploadedImageUrl, author: currentUserEmail // Envoie l'e-mail de la session actuelle }]); if (error) { // S'il y a un problème (ex: colonne 'author' manquante), l'erreur s'affiche ici alert("Erreur lors de l'enregistrement : " + error.message); } else { document.getElementById('recipeForm').reset(); showPage('page-accueil'); } });
